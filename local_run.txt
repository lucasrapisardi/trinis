# 1. Copy env and fill in your keys
cp .env.example .env

# 2. Generate a Fernet key for token encryption
python -c "from cryptography.fernet import Fernet; print(Fernet.generate_key().decode())"
# → paste into SHOPIFY_TOKEN_ENCRYPTION_KEY

# 3. Start Postgres + Redis
docker-compose up db redis -d

# 4. Install deps
pip install -r requirements.txt

# 5. Start the API
uvicorn app.main:app --reload
# → http://localhost:8000/docs

# 6. Start a Celery worker (separate terminal)
celery -A app.tasks.celery_app worker --loglevel=info -Q scrape,enrich,image,sync,default
